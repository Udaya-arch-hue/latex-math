\begin{enumerate}

\item[1985/1.]
A circle has center on the side $AB$ of the cyclic quadrilateral $ABCD$.

The other three sides are tangent to the circle.

Prove that

\begin{align*}
AD+BC=AB
\end{align*}

\vspace{0.5cm}

\item[1985/2.]
Let $n$ and $k$ be given relatively prime natural numbers, $k<n$.

Each number in the set

\begin{align*}
M=\{1,2,\ldots,n-1\}
\end{align*}

is colored either blue or white.

It is given that

\begin{enumerate}
\item[(i)]
for each $i\in M$, both $i$ and $n-i$ have the same color;

\item[(ii)]
for each $i\in M$, $i\neq k$, both $i$ and $|i-k|$ have the same color.
\end{enumerate}

Prove that all numbers in $M$ must have the same color.

\vspace{0.5cm}

\item[1985/3.]
For any polynomial

\begin{align*}
P(x)=a_0+a_1x+\cdots+a_qx^q
\end{align*}

with integer coefficients, the number of coefficients which are odd is denoted by $w(P)$.

For

\begin{align*}
i=0,1,\ldots,t,
\end{align*}

let

\begin{align*}
Q_i(x)=(1+x)^i
\end{align*}

Prove that if

\begin{align*}
0\leq i_1<i_2<\cdots<i_n
\end{align*}

then

\begin{align*}
w(Q_{i_1}+Q_{i_2}+\cdots+Q_{i_n})\geq w(Q_{i_n})
\end{align*}

\vspace{0.5cm}

\item[1985/4.]
Given a set $M$ of $1985$ distinct positive integers, none of which has a prime divisor greater than $26$.

Prove that $M$ contains at least one subset of four distinct elements whose product is the fourth power of an integer.

\vspace{0.5cm}

\item[1985/5.]
A circle with center $O$ passes through the vertices $A$ and $C$ of triangle $ABC$ and intersects the segments $AB$ and $BC$ again at distinct points $K$ and $N$, respectively.

The circumscribed circles of the triangles $ABC$ and $KBN$ intersect at exactly two distinct points $B$ and $M$.

Prove that angle $OMB$ is a right angle.

\vspace{0.5cm}

\item[1985/6.]
For every real number $x_1$, construct the sequence

\begin{align*}
x_1,x_2,\ldots
\end{align*}

by setting

\begin{align*}
x_{n+1}=x_n\left(x_n+\frac{1}{n}\right)
\qquad \text{for each } n\geq 1
\end{align*}

Prove that there exists exactly one value of $x_1$ for which

\begin{align*}
0<x_n<x_{n+1}<1
\end{align*}

for every $n$.

\end{enumerate}
